int rand_0(int n)

int main()
{
	int i, n, r;

	//sets seed for the random number
	srand(time(NULL)); //NULL MUST BE CAPITALIZED

	cout << "Enter how many dice will you roll: ";
	cin >> n;

	//initialization
		//condition
			//increment
	for (i=1; i <= n; i++)
	{
	r = rand_0(6) + 1;     //rand() % 6 + 1;  // number will be 0-5, use the +1 so it will be 1-6 like a die
	cout << r << endl;
	}




system ("pause");

}


int rand_0(int n)
{
return rand() % n;
}
